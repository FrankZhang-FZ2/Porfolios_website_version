MLM

#MLM clean na
MLM <- MLM %>%
  filter(Ext != "NA")

##centering
#Center Ext
MLM <- MLM %>%
  mutate(cext = Ext - mean(Ext))

#center level 1 variables
MLM <- MLM %>%
  group_by(Person) %>%
  mutate(msi = mean(NumPeople),
         csi = NumPeople - msi)

##analyses 
m_MLM <- lmerTest::lmer(Hap ~ cext + csi + cext*csi + (csi + 1 || Person) , data = MLM)
